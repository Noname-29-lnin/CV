	\section{Projects}
	
	\begin{twocolentry}{
		\textit{\href{https://github.com/Noname-29-lnin/Viterbi_decoder.git}{Noname-29-lnin\\/Viterbi\_decoder}}}
		\textbf{Design and Implementation of a Power-Efficient Viterbi Encoding and Decoding Architecture on FPGA: From Theory to Practical Application}
	\end{twocolentry}
	
	\vspace{0.10 cm}
	\begin{onecolentry}
		\begin{highlights}
			\item This project involves the design and implementation of a \textbf{Viterbi decoder} for convolutional codes with parameters $(3, 1, 2)$ (constraint length $K=3$, code rate $1/2$) on an FPGA. The Viterbi algorithm is optimized for error correction in noisy communication channels, leveraging trellis-based path metrics and survivor path selection. The decoder is structured using key functional blocks, including the \textit{Branch Metric Unit (BMU)}, \textit{Add-Compare-Select Unit (ACSU)}, \textit{Path Metric Unit (PMU)}, and \textit{Survivor Path Memory Unit (SPMU)}.
			
			\item To test the decoder, additional modules such as a \textit{Parallel-In-Serial-Out (PISO)} register, \textit{Serial-In-Parallel-Out (SIPO)} register, and \textit{LCD IP core} (for hardware-based LCD control) are integrated. These components enable practical verification of the Viterbi decoder's functionality. The final implementation is deployed on a \textit{DE2 development kit}, where the design is synthesized and programmed onto the FPGA for real-world validation.
			\item Tools \& Lauguage:
				\begin{itemize}[label=-]
					\item Simulation: Verilator, GTKwave.
					\item Synthesis: Intel Quartus Prime.
					\item Language: C, SystemVerilog.
				\end{itemize}
		\end{highlights}
	\end{onecolentry}
	
	\begin{twocolentry}{
			\textit{\href{https://github.com/Noname-29-lnin/Sort_on_FPGA.git}{Noname-29-lnin \\ /Sort\_FPGA}}}
		\textbf{FPGA-Based Sorting Using Independent Subarrays of Approximately Equal Length}
	\end{twocolentry}
	\vspace{0.10 cm}
	\begin{onecolentry}
		\begin{highlights}
			\item Designed and implemented the hardware architecture for a high-throughput sorting framework, based on the research \emph{"A General Framework for Sorting Large Data Sets Using Independent Subarrays of Approximately Equal Length"}.
			\item The RTL implementation targets performance optimization for large-scale datasets (from $2^{18}$ to $2^{20}$+ elements), supporting multiple data types.
			\item Engineered the core processing unit to handle independent subarrays, optimizing for either area (serial processing) or throughput (parallel processing) to balance resource utilization and sorting speed on the FPGA.
			\item \textbf{Tools \& Technologies:}
			\begin{itemize}[label=-]
				\item \textbf{Simulation:} Verilator, QuestaSim, GTKWave.
				\item \textbf{Synthesis \& Implementation:} Intel Quartus Prime.
				\item \textbf{Languages:} SystemVerilog (RTL Design), C++, Python.
			\end{itemize}
		\end{highlights}
